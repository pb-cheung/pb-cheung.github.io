# 生命周期

# 生命周期有哪些

  • beforeCreate，这里没有实现响应式。初始化父子关系$parent,$children。vue 3中不再使用了
  • created，拿到的是响应式的属性（不涉及到dom渲染），可以在服务端渲染中使用
  • beforeMount，意义不大，基本不用
  • mounted，渲染完成，可以获取真实DOM $el
  • beforeUpdate，组件更新前执行，flushSchedulerQueue 调度器清空队列函数中，watcher.run开始执行前
  • updated，watcher.run执行后调用
  • beforeDestroy，卸载dep和watcher、销毁$children前调用。
  • destroyed 销毁后触发
  • errorCaptured 捕获错误

# beforeCreate&created

这两个事件在src/core/instance/lifecycle文件，初始化函数_init中调用。

源码：

Vue.prototype._init = function (options?: Object) {
  // ...
  initLifecycle(vm) // 初始化父子关系
  initEvents(vm) // 初始化事件（$on/$once/$off）
  initRender(vm)  // 声明_c/_v/_s，$slot这些函数

  callHook(vm, 'beforeCreate')

  initInjections(vm) // resolve injections before data/props
  initState(vm) // 添加响应式
  initProvide(vm) // resolve provide after data/props
  callHook(vm, 'created')
  // ...
}
  • beforeCreate在initState之前，data、methods、props、computed、watch都获取不到
  • 需要访问数据的逻辑放到created

# beforeMounted&mounted

这两个钩子函数在src/core/instance/lifecycle文件，初始化函数mountComponent中调用触发。

export function mountComponent(
  vm: Component,
  el: Element | null | undefined,
  hydrating?: boolean
): Component {
  vm.$el = el
  if (!vm.$options.render) {
    // @ts-expect-error invalid type
    vm.$options.render = createEmptyVNode
    if (__DEV__) {
      /* istanbul ignore if */
      if (
        (vm.$options.template && vm.$options.template.charAt(0) !== '#') ||
        vm.$options.el ||
        el
      ) {
        warn(
          'You are using the runtime-only build of Vue where the template ' +
            'compiler is not available. Either pre-compile the templates into ' +
            'render functions, or use the compiler-included build.',
          vm
        )
      } else {
        warn(
          'Failed to mount component: template or render function not defined.',
          vm
        )
      }
    }
  }
  callHook(vm, 'beforeMount')

  let updateComponent
  /* istanbul ignore if */
  if (__DEV__ && config.performance && mark) {
    updateComponent = () => {
      const name = vm._name
      const id = vm._uid
      const startTag = `vue-perf-start:${id}`
      const endTag = `vue-perf-end:${id}`

      mark(startTag)
      const vnode = vm._render()
      mark(endTag)
      measure(`vue ${name} render`, startTag, endTag)

      mark(startTag)
      vm._update(vnode, hydrating)
      mark(endTag)
      measure(`vue ${name} patch`, startTag, endTag)
    }
  } else {
    updateComponent = () => {
      vm._update(vm._render(), hydrating)
    }
  }

  const watcherOptions: WatcherOptions = {
    before() {
      if (vm._isMounted && !vm._isDestroyed) {
        callHook(vm, 'beforeUpdate')
      }
    }
  }

  if (__DEV__) {
    watcherOptions.onTrack = e => callHook(vm, 'renderTracked', [e])
    watcherOptions.onTrigger = e => callHook(vm, 'renderTriggered', [e])
  }

  // we set this to vm._watcher inside the watcher's constructor
  // since the watcher's initial patch may call $forceUpdate (e.g. inside child
  // component's mounted hook), which relies on vm._watcher being already defined
  new Watcher(
    vm,
    updateComponent,
    noop,
    watcherOptions,
    true /* isRenderWatcher */
  )
  hydrating = false

  // flush buffer for flush: "pre" watchers queued in setup()
  const preWatchers = vm._preWatchers
  if (preWatchers) {
    for (let i = 0; i < preWatchers.length; i++) {
      preWatchers[i].run()
    }
  }

  // manually mounted instance, call mounted on self
  // mounted is called for render-created child components in its inserted hook
  if (vm.$vnode == null) {
    vm._isMounted = true
    callHook(vm, 'mounted')
  }
  return vm
}
  • 在执行vm._render()函数渲染VNode之前，执行beforeMounthook。
  • 在执行完vm._update() 把VNodepatch到真实DOM后，执行 mountedhook。需要操作DOM的逻辑放到这个hook里开始执行。
  • 子组件mounted先于父组件mounted

# 父子组件的差异

  • vm.$vnode == null用来判断当前是否是根组件，$vnode属性指向父组件的vnode，根组件没有父组件为初始值null。
  • Vue中组件渲染是深度优先递归渲染的，渲染一个组件时，如果它有子组件，则先渲染子组件。
  • 根组件mounted在mountComponent函数，子组件mounted通过其patch => invokeInsertHook => componentVNodehooks.insert。
  • patch函数中的变量insertedVnodeQueue的主要作用是记录子组件VNode的插入顺序（该变量在每个子组件$mount后用initComponent中插入当前VNode）
  • invokeInsertHook函数中，依次获取调用insertedVnodeQueue维护的子组件VNode的inser钩子函数，此钩子函数中完成组件mountedhook触发
return function patch() {
    // ...
    invokeInsertHook(vnode, insertedVnodeQueue, isInitialPatch)
    return vnode.elm
}

function invokeInsertHook(vnode, queue, initial) {
    // delay insert hooks for component root nodes, invoke them after the
    // element is really inserted
    if (isTrue(initial) && isDef(vnode.parent)) {
      vnode.parent.data.pendingInsert = queue
    } else {
      for (let i = 0; i < queue.length; ++i) {
        queue[i].data.hook.insert(queue[i])
      }
    }
  }

const componentVNodeHooks = {
  // ...
  insert (vnode: MountedComponentVNode) {
    const { context, componentInstance } = vnode
    if (!componentInstance._isMounted) {
      componentInstance._isMounted = true
      callHook(componentInstance, 'mounted')
    }
    // ...
  },
}

# beforeUpdate&update

  • mountComponent函数中会创建渲染渲染watcher实例，beforeUpdate钩子在watcher实例的before函数中触发
  • flushSchedulerQueue中先出发before => beforeUpdate，再调用callUpdatedHooks => updatehook。
export function mountComponent (
  vm: Component,
  el: ?Element,
  hydrating?: boolean
): Component {
  // ...
  // 这里是简写
  let updateComponent = () => {
      vm._update(vm._render(), hydrating)
  }
  new Watcher(vm, updateComponent, noop, {
    before () {
      if (vm._isMounted) {
        callHook(vm, 'beforeUpdate')
      }
    }
  }, true /* isRenderWatcher */)
  // ...
}

function flushSchedulerQueue() {
     watcher.before()

     callUpdatedHooks()
}
function callUpdatedHooks(queue: Watcher[]) {
  let i = queue.length
  while (i--) {
    const watcher = queue[i]
    const vm = watcher.vm
    if (vm && vm._watcher === watcher && vm._isMounted && !vm._isDestroyed) {
      callHook(vm, 'updated')
    }
  }
}

# beforeDestroyed&destroyed

  • 删除父子关系，删除watcher
  • destroyedhook触发顺序和mounted一样：先子组件再父组件。调用之前先调用vm.__patch__(vm._vnode, null)触发子组件的销毁钩子函数
  Vue.prototype.$destroy = function () {
    const vm: Component = this
    if (vm._isBeingDestroyed) {
      return
    }
    callHook(vm, 'beforeDestroy')
    vm._isBeingDestroyed = true
    // remove self from parent
    const parent = vm.$parent
    if (parent && !parent._isBeingDestroyed && !vm.$options.abstract) {
      remove(parent.$children, vm)
    }
    // teardown scope. this includes both the render watcher and other
    // watchers created
    vm._scope.stop()
    // remove reference from data ob
    // frozen object may not have observer.
    if (vm._data.__ob__) {
      vm._data.__ob__.vmCount--
    }
    // call the last hook...
    vm._isDestroyed = true
    // invoke destroy hooks on current rendered tree
    vm.__patch__(vm._vnode, null)
    // fire destroyed hook
    callHook(vm, 'destroyed')
    // turn off all instance listeners.
    vm.$off()
    // remove __vue__ reference
    if (vm.$el) {
      vm.$el.__vue__ = null
    }
    // release circular reference (#6759)
    if (vm.$vnode) {
      vm.$vnode.parent = null
    }
  }
}

# 在哪一步发送请求及原因

一般在mounted（created也可以）

代码是同步执行的，请求是异步的。

created -> mounted 这个过程是异步的，即使在created回调函数中发送请求， 请求回调函数也一定是在mounted之后。（事件循环机制——Ajax的回调函数会放入到宏任务队列）
