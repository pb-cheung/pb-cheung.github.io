# 二、编译过程

export function compileToFunction(template) {
    // 1.将template转换成AST语法树
    const ast = parseHTML(template);
    // console.log('AST: ', ast);

    // 2.生成render方法（render方法返回的结果就是虚拟DOM）
    // 模板引擎的实现原理一般都是 with + new Function
    let code = codegen(ast);
    code = `with(this) { return ${code}; }`;
    let render = new Function(code); // 根据代码生成render函数

    // 可以看到render是一个匿名函数，with的参数this是render函数的调用方
    return render;
}

# 将模板转换成抽象语法树

AST语法树的样子：

数据结构是：

  attrs: [

      name: "style",

# parseHTML函数

parseHTML是转换函数，接收模板字符串，返回AST语法树对象

export default function parseHTML(html) {
  const stack = []; // 栈用于存储元素
  let currentParent; // 指向“栈顶”元素
  let root;

  function createASTElement(tag, attrs) {} // 返回AST单个节点
  function start(tag, attrs) {}   // 利用栈数据结构来构造树
  function chars(text) {} // 创建文本节点并放入树上
  function end(tag) {} // 出栈：栈顶元素弹出，移动栈顶指针
  function advance(n) {}   // 前进：删除已经匹配的字符
  function parseStartTag() {} // 解析开始标签：获取标签名、属性

  while (html) {
        // textEnd为0 => 说明是一个开始标签或者结束标签
        // textEnd>0 => 说明是文本的结束位置
        let textEnd = html.indexOf('<');
        if (textEnd === 0) { // 标签内容：标签名、属性
            const startTagMatch = parseStartTag(html); // 开始标签 的匹配结果
            if (startTagMatch) {
                start(startTagMatch.tagName, startTagMatch.attrs);
                continue;
            }
            const endTagMatch = html.match(endTag);
            if (endTagMatch) {
                advance(endTagMatch[0].length);
                end(endTagMatch[1]);
                continue;
            }
        }
        if (textEnd > 0) { // 截取文本
            let text = html.substring(0, textEnd); // 文本内容
            if (text) {
                advance(text.length);
                chars(text);
            }
        }
    }
  return root;
}

# 函数主要流程：

  • while循环遍历模板字符串，结束条件是模板字符串为空，
  • 循环体为一个条件判断分支结构
    • 是开始标签 => 解析开始标签（获取标签名和属性） => 构建AST节点；
    • 是关闭标签 => 获取文本内容

# AST的节点的数据结构：

ASTElement: {
  tag, // 标签名
  type, // 类型，标签or文本
  attrs: [
    {
        name: 'xxx',
        value: 'abc'
    }
  ], // 标签属性
  children: [ASTElement, ... ], // 子元素
  parent: null // 父元素
}

# 构建AST

利用栈数据结构构建抽象语法树

  • 开始标签 => 创建节点 => 入栈 + 设置父子属性
  • 关闭标签 => 出栈
function start(tag, attrs) {
  // console.log('开始：', tag, attrs);
  let node = createASTElement(tag, attrs); // 创建一个AST节点
  if (!root) {
    root = node;
  }
  if (currentParent) {
    node.parent = currentParent;
    currentParent.children.push(node);
  }
  stack.push(node); // 入栈
  currentParent = node;
}

function end(tag) {
  stack.pop(); // 出栈
  currentParent = stack[stack.length - 1];
}

# 代码生成函数

根据节点类型代码生成操作拆分出许多函数：

  • codegen入口函数，接收ast对象，返回字符串;
  • genProps属性处理函数，遍历每个节点的标签数组，拼接属性字符串key1:val1,ke2:val2;
  • genChildren遍历子节点，每个子节点调用gen函数
  • gen根据节点类型分支，元素类型调用codegen（递归），文本节点执行对应的代码生成逻辑

入口函数：

function codegen(ast) {
    let code = `_c('${ast.tag}',
    ${ast.attrs.length > 0 ? genProps(ast.attrs): 'null'},
    ${ast.children.length ? genChildren(ast): 'null'})`;
    return code;
}

# 生成规则

代码生成函数生成代码的规则是什么？

元素类型，生成为_c(标签名, 属性, 子元素)函数 文本类型，生成为_v(${JSON.stringify(text)})函数 包含插值的文本，生成为_s()函数

# 修改this

with + new Function()

从上文打印的render函数示例，我们可以看出：

  • render函数是一个匿名函数
  • 函数体为with表达式，接收参数为this，这个this指向匿名函数的调用方（我们暂且把它叫做C）

根据with的用法，其参数(this指向的C)被添加到花括内号包裹的语句变量的作用域链。那么_c、_v、_s这些函数未找到定义的函数就沿着作用域链线上查找。

说道这里，模板编译的过程已经结束了，但是要明白代码生成为什么要这样设计，还需要继续了解render函数的调用。

# render函数调用

render函数是模板编译生成的产物。

render函数的调用者就是Vue.prototype（上文C）：

Vue.prototype._render = function () {
    // 当渲染的时候会去实例中取值，我们就可以将属性和视图绑定在一起
    return this.$options.render.call(this);
}

所以，_c、_v、_s也在Vue.prototype上定义。

这么做的目的就是：模板中使用变量无需用点号运算符访问（object.property），直接用变量名即可，且变量从当前vue实例中查找。
