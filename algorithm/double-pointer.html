# 双指针

# 一、题目

力扣 977 有序数组的平方 (opens new window)

给你一个按 非递减顺序 排序的整数数组 nums，返回 每个数字的平方 组成的新数组，要求也按 非递减顺序 排序。
示例 1：
输入：nums = [-4,-1,0,3,10]
输出：[0,1,9,16,100]
解释：平方后，数组变为 [16,1,0,9,100]
排序后，数组变为 [0,1,9,16,100]
示例 2：
输入：nums = [-7,-3,2,3,11]
输出：[4,9,9,49,121]
提示：
1 <= nums.length <= 104
-104 <= nums[i] <= 104
nums 已按 非递减顺序 排序
进阶：
请你设计时间复杂度为 O(n) 的算法解决本问题

# 二、双指针法

  • 利用好题设所有条件，非递减数组。
  • 因为可能存在负数、正数的情况，负数值越大平方越小，正数值越大平方越大，不能遍历用一个逻辑从头到尾。
  • 两个指针分别指向负数、正数，具体又可以分为两种
    • 指向正负数分解处的两个索引，然后分别向两端走
    • 一个指向数组开头索引，一个指向数组末尾，相向而走
  • 剩下就是两个指针指向元素的平方，比较大小，归并放入结果数组

代码：

/**
 * @param {number[]} nums
 * @return {number[]}
 */
var sortedSquares = function(nums) {
    let result = [];
    const len = nums.length;
    let pNegative = 0;
    let pPositive = 0;
    if (nums[0] >= 0) { // 非负数数组
        result = nums.map(ele => Math.pow(ele, 2));
        return result;
    }
    if (nums[len - 1] <= 0) { // 非正数，需要逆序添加
        for (let i = len - 1; i >= 0; i--) {
            result.push(Math.pow(nums[i], 2));
        }
        return result;
    }
    // 负数、正数都有，找正负数的分界线
    for (let i = 1; i < len; i++) {
        if (nums[i] >= 0) {
            pNegative = i - 1;
            pPositive = i;
            break;
        }
    }
    console.log(pNegative, pPositive)
    // 数组二合一
    while (pNegative >= 0 || pPositive <= len - 1) {
        if (pNegative < 0) { // 后侧还有剩余
            const val = Math.pow(nums[pPositive], 2);
            result.push(val);
            pPositive ++;
            continue;
        }
        if (pPositive > len - 1) { // 左侧还有剩余
            const val = Math.pow(nums[pNegative], 2);
            result.push(val);
            pNegative --;
            continue;
        }
        const val1 = Math.pow(nums[pNegative], 2);
        const val2 = Math.pow(nums[pPositive], 2);
        if (val1 < val2) {
            result.push(val1);
            pNegative --;
        } else {
            result.push(val2);
            pPositive ++;
        }
    }
    return result;
};

# 三数和

https://leetcode.cn/problems/3sum/

给你一个整数数组 nums ，判断是否存在三元组 [nums[i], nums[j], nums[k]] 满足 i != j、i != k 且 j != k ，同时还满足 nums[i] + nums[j] + nums[k] == 0 。请你返回所有和为 0 且不重复的三元组。 注意：答案中不可以包含重复的三元组。

示例 1：

输入：nums = [-1,0,1,2,-1,-4]
输出：[[-1,-1,2],[-1,0,1]]
解释：
nums[0] + nums[1] + nums[2] = (-1) + 0 + 1 = 0 。
nums[1] + nums[2] + nums[4] = 0 + 1 + (-1) = 0 。
nums[0] + nums[3] + nums[4] = (-1) + 2 + (-1) = 0 。
不同的三元组是 [-1,0,1] 和 [-1,-1,2] 。
注意，输出的顺序和三元组的顺序并不重要。

// 双指针O(n²)
var threeSum = function(nums) {
    let result = [];
    nums.sort((a,b) => a - b);
    for (let i = 0; i < nums.length - 2; i++) {
        if (nums[i] > 0) break; // 第一个数大于0，则三数相加必然大于0，不符合
        if(i > 0 && nums[i] === nums[i-1]) continue; // 去重
        let L = i + 1;
        let R = nums.length - 1;
        while(L < R) {
            const sum = nums[L] + nums[i] + nums[R];
            if (sum === 0) {
                result.push([nums[L], nums[i], nums[R]]);
                while(L < R && nums[L] === nums[L + 1]) L++; // 去重（相邻相同元素）
                while(L < R && nums[R] === nums[R - 1]) R--; // 去重
                L ++;
                R --;
            } else if (sum < 0) {
                L ++;
            } else if (sum > 0) {
                R --;
            }
        }
    }
    return result;
}

const nums1 = [-1,0,1,2,-1,-4];
console.log(threeSum(nums1));

const nums2 = [0,1,1];
console.log(threeSum(nums2));

const nums3 = [0,0,0];
console.log(threeSum(nums3));

const nums4 = [-1,0,1,2,-1,-4];
console.log(threeSum(nums4));

const nums5 = [0,0,0];
console.log(threeSum(nums5))

# 盛最多水的容器

盛最多水的容器 (opens new window)

给定一个长度为 n 的整数数组 height 。有 n 条垂线，第 i 条线的两个端点是 (i, 0) 和 (i, height[i]) 。 找出其中的两条线，使得它们与 x 轴共同构成的容器可以容纳最多的水。 返回容器可以储存的最大水量。 说明：你不能倾斜容器。

/**
 * @param {number[]} height
 * @return {number}
 */
// 双指针
var maxArea = function(height) {
    const result = new Array(height.length).fill(-100);
    let left = 0;
    let right = height.length - 1;
    while(left < right) {
        if (height[left] < height[right]) {
            result[left] = height[left] * (right - left);
            left ++;
        } else if (height[left] > height[right]) {
            result[right] = height[right] * (right - left);
            right --;
        } else {
            result[left] = height[left] * (right - left);
            result[right] = height[right] * (right - left);
            left ++;
            right --;
        }
    }
    // console.log(result)
    result.sort((a,b) => a - b);
    return result[result.length - 1];
};

console.log(maxArea(height1))

const height2 = [0,2];
console.log(maxArea(height2));

console.log(maxArea(heigth3));
