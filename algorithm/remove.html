/**
 * 力扣27.移除元素
 * https://leetcode.cn/problems/remove-element/description/
 * @param {number[]} nums
 * @param {number} val
 * @return {number}
 */
var removeElement = function(nums, val) {
    let k = 0
    let len = 0
    for(let i = 0; i < nums.length; i ++) {
        if (nums[i] !== val) {
            nums[k++] = nums[i];
            len ++;
        }
    }
    nums.slice(0, len)
    console.log(nums)
    return len;
};

/**
 * 力扣 26. 删除有序数组中的重复项
 * @param {number[]} nums
 * @return {number}
 */
var removeDuplicates = function(nums) {
    let val = nums[0];
    let k = 1;
    let len = 1;
    for (let i = 1; i < nums.length; i ++) {
        val = nums[i - 1];
        if (nums[i] !== val) {
            nums[k++] = nums[i];
            len ++;
            console.log(nums);
        }
    }
    nums = nums.slice(0, len);
    return len;
};

/**
 * 力扣 283. 移动零
 * @param {number[]} nums
 * @return {void} Do not return anything, modify nums in-place instead.
 */
var moveZeroes = function(nums) {
    let target = 0;
    let fast = 0; //
    let low = 0; // 慢指针左侧不包含0
    let len = 0;
    function swap(arr, x, y) {
        let temp = arr[x];
        arr[x] = arr[y];
        arr[y] = temp;
    }
    while(fast < nums.length) {
        if (nums[fast] !== target) {
            swap(nums, low, fast);
            low ++;
            len ++;
        }
        fast++;
    }
    return nums;
};

const nums = [0,1,0,3,12]

console.log(moveZeroes(nums))
