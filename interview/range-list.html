# 答案源码

export default class RangeList {
    constructor() {
        this.list = []
    }
    /**
     * Adds a range to the list
     * @param {Array<number>} range - Array of two integers that specify
     * beginning and end of range.
     */
    add(range) {
        this.list.push(range)
        /**
         * Merges all overlapping ranges in list
         * @param {Array<range>} list - Array of ranges
         * @return {Array<range>} Array of ranges
         */
        function mergeRange(list) {
            list.sort((a, b) => a[0] - b[0])
            let prev = list[0]
            let result = []
            for(let i = 0; i < list.length; i++) {
                let cur = list[i]
                if(cur[0] > prev[1]) {
                    result.push(prev)
                    prev = cur
                } else {
                    prev[1] = Math.max(cur[1], prev[1])
                }
            }
            result.push(prev)
            return result
        }
        this.list = mergeRange(this.list)
    }
    /**
     * Removes a range from the list
     * @param {Array<number>} range - Array of two integers that specify
     * beginning and end of range.
     */
    remove(range) {
        const [removeStart, removeEnd] = range
        let result = []
        for (let i = 0; i < this.list.length; i++ ) {
            const currRange = this.list[i];
            const [currStart, currEnd] = currRange;
            if (currStart >= removeEnd || currEnd < removeStart) { // not in current range
                result.push(currRange)
            } else {
                if (currStart < removeStart) {
                    result.push([currStart, removeStart])
                }
                if (currEnd > removeEnd) {
                    result.push([removeEnd, currEnd])
                }
            }
        }
        this.list = result
    }
    /**
     * Prints out the list of ranges in the range list
     */
    print() {
        /**
         * Format range list to string.  [[1, 3], [19, 21]] => `'[1, 3) [19, 21)'`
         * @param {Array<range>} list - Array of ranges
         * @return {string} - formatted range list string like: `'[1, 3) [19, 21)'`
         */
        function format(list) {
            let resultStr = ''
            list.forEach(range => {
                const [start, end] = range
                resultStr += `[${start}, ${end}) `
            })
            resultStr = resultStr.trim()
            return resultStr
        }
        console.log(format(this.list))
    }
}
